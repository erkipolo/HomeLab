# Apt via Proxy
tee -a /etc/apt/apt.conf <<EOF
acquire::http::proxy::localhost "DIRECT";
acquire::socks::proxy           "http://localhost:3128";
acquire::https::proxy           "http://localhost:3128/";
acquire::http::proxy            "http://localhost:3128/";
acquire::ftp::proxy             "http://localhost:3128";
EOF

# Apt no check valid date
tee -a /etc/apt/apt.conf <<EOF
acquire::check-valid-until "0";
EOF

# Enable method FTP
tee -a /etc/apt/apt.conf <<EOF
dir::bin::methods::ftp "ftp";
EOF

# Borramos la cache
apt clean

# Borramos la cache de archivos
apt autoclean

# Actualizamos la cache
apt update

# Actualizamos el sistema
apt -yq dist-upgrade

# Removemos los paquetes huerfanos de el sistema
apt -yq --purge autoremove
