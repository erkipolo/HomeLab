# Servicio que se emplea para cachear los
# paquetes que se instalan en los ordenadores

# Instalamos la aplicaci√≥n
apt -yq install apt-cacher

# C0nfiguramos el servicio
tee /etc/apt-cacher/apt-cacher.conf <<EOF
cache_dir = /var/cache/apt-cacher
log_dir = /var/log/apt-cacher
daemon_port = 3142
group = www-data
user = www-data
daemon_addr = localhost, 0.0.0.0
http_proxy = 192.168.0.1:3128
use_proxy = 1
allowed_hosts = *
EOF

# Reiniciamos el servicio
systemctl restart apt-cacher

# Chequeamos el estado de el servicio
systemctl status apt-cacher
